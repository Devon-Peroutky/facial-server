CREATE TABLE stars (
	name varchar(255) PRIMARY KEY,
	bio text,
	twitter varchar(255),
	website varchar(255),
	recent_work varchar(255),
	white tinyint(1),
	black tinyint(1),
	latina tinyint(1),
	asian tinyint(1)
)ENGINE=InnoDB;

CREATE TABLE images (
	name varchar(255),
	img varchar(255),
	face_id varchar(255),
	FOREIGN KEY (name) REFERENCES stars (name),
	PRIMARY KEY (name, img)
)ENGINE=InnoDB;
